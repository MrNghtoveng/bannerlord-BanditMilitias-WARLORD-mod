        private static int FindSlotIndex(MobileParty party, CharacterObject character)
        {
            for (int i = 0; i < party.MemberRoster.Count; i++)
            {
                if (party.MemberRoster.GetElementCopyAtIndex(i).Character == character)
                    return i;
            }
            return -1;
        }
