        private void OnMilitiaKilled(MilitiaKilledEvent evt)
        {
            if (!evt.WasPlayerKill) return;
            if (evt.HomeHideout == null) return;

            var warlord = WarlordSystem.Instance.GetWarlordForHideout(evt.HomeHideout);
            if (warlord == null || !warlord.IsAlive) return;

            if (!_voices.TryGetValue(warlord.Personality, out var voice)) return;

            _ = _playerKillsPerWarlord.TryGetValue(warlord.StringId, out int kills);
            kills++;
            _playerKillsPerWarlord[warlord.StringId] = kills;

            string playerName = Hero.MainHero?.Name?.ToString() ?? "stranger";
            string? msg = voice.GetKillReaction(warlord, playerName, kills);
            if (msg != null)
                NarrativePresenter.Show(msg, warlord.Personality, kills);
        }

        private void OnRaidCompleted(MilitiaRaidCompletedEvent evt)
        {
            if (evt.RaiderParty == null || evt.TargetVillage == null) return;

            var warlord = WarlordSystem.Instance.GetWarlordForParty(evt.RaiderParty);
            if (warlord == null) return;

            string? msg = voice.GetRaidEcho(warlord, villageName);

        private void OnLevelChanged(WarlordLevelChangedEvent evt)
        {
            if (evt.Warlord == null) return;
            if (!_arcTexts.TryGetValue(evt.NewLevel, out var arc)) return;

            var warlord = evt.Warlord;
            bool isPromotion = evt.NewLevel > evt.OldLevel;

            string voiceLine = string.Empty;
            if (_voices.TryGetValue(warlord.Personality, out _))
            {
                voiceLine = warlord.Personality switch
                {

                    PersonalityType.Aggressive => $"[{warlord.Name}]: 'This power is mine now. I want more.'",

                    PersonalityType.Cautious => $"[{warlord.Name}]: 'The people know me. This is just the beginning.'",
                    _ => string.Empty,
                };

            string playerMsg = arc.PlayerMsg;
            string worldMsg = arc.WorldMsg;

            if (isPromotion)
            {
                string fullText = $"{playerMsg}\n\n{worldMsg}";
                if (!string.IsNullOrEmpty(voiceLine))
                    fullText += $"\n\n{voiceLine}";

                NarrativePresenter.ShowLegitimacyArc(

            {

                NarrativePresenter.Show(demotionMsg, warlord.Personality);
            }
        }

        private NarrativeMemory GetOrCreateMemory(string warlordId)
        {
            if (!_warlordMemories.TryGetValue(warlordId, out var mem))

        public override void OnDailyTick()
        {
            if (!_isInitialized) return;

            try
            {
                var warlords = WarlordSystem.Instance.GetAllWarlords()
                    .Where(w => w.IsAlive && w.DaysActive > 5)
                    .ToList();

                if (warlords.Count == 0) return;

                foreach (var w in warlords)
                {
                    var mem = GetOrCreateMemory(w.StringId);

                    // Tension escalation check
                    if (mem.ShouldEscalate())
                    {
                        mem.EscalateTension();
                        // When tension rises, always generate rumor
                        string tensionMsg = WorldVoiceSynthesizer.GenerateAmbientLine(w, mem);
                        NarrativePresenter.ShowAmbient(tensionMsg);
                    }
                    else if (MBRandom.RandomFloat < 0.12f)
                    {
                        // Normal daily organic message chance
                        string msg = WorldVoiceSynthesizer.GenerateAmbientLine(w, mem);
                        NarrativePresenter.ShowAmbient(msg);
                    }
                }
            }
            catch (Exception ex)
            {
                BanditMilitias.Debug.DebugLogger.Warning("NarrativeSystem", $"OnDailyTick Error: {ex.Message}");
            }
        }

    public class NarrativeMemory
    {
        private readonly Dictionary<string, int> _templateFatigue = new();
        private readonly Queue<string> _recentEvents = new();
        private const int MAX_RECENT_MEMORY = 8;

        public int DaysSinceEscalation { get; private set; } = 0;
        public int TensionLevel { get; private set; } = 0; // 0-5

        // Yeni bir olay kaydedildiÄŸinde
        public void RecordEvent(string templateKey, int currentDay)
        {
            if (_templateFatigue.TryGetValue(templateKey, out int count))
                _templateFatigue[templateKey] = count + 1;
            else
                _templateFatigue[templateKey] = 1;

            _recentEvents.Enqueue(templateKey);
            if (_recentEvents.Count > MAX_RECENT_MEMORY)
                _ = _recentEvents.Dequeue();

            // Her olay tansiyonu sÄ±fÄ±rlar veya hafifletir
            DaysSinceEscalation++;
        }

        // Bir ÅŸablonun "yorgunluk" (ne kadar kullanÄ±ldÄ±) skorunu getirir. Max %90
        public float GetFatigue(string templateKey)
        {
            if (!_templateFatigue.TryGetValue(templateKey, out int count))
                return 0f;

            return Math.Min(0.9f, count * 0.15f);
        }

        // Son 3 olayda geÃ§ip geÃ§mediÄŸini kontrol eder
        public bool IsRecentlyUsed(string templateKey)
        {
            return _recentEvents.Reverse().Take(3).Contains(templateKey);
        }

        public bool ShouldEscalate() => DaysSinceEscalation > 6;

        public void EscalateTension()
        {
            TensionLevel = Math.Min(5, TensionLevel + 1);
            DaysSinceEscalation = 0;
        }

        public void ResetTension()
        {
            TensionLevel = 0;
            DaysSinceEscalation = 0;
        }

        // AÄŸÄ±rlÄ±klÄ± rastgele seÃ§im motoru (YorulmuÅŸ cÃ¼mleleri cezalandÄ±rarak en tazesini seÃ§er)
        public T WeightedPick<T>(IEnumerable<T> candidates, Func<T, string> keySelector)
        {
            var pool = new List<T>();
            foreach (var candidate in candidates)
            {
                string key = keySelector(candidate);
                float fatigue = GetFatigue(key);
                bool isRecent = IsRecentlyUsed(key);

                int baseWeight = 4;
                int recentPenalty = isRecent ? -3 : 0;
                int fatiguePenalty = (int)(fatigue * 4); // max -3

                int finalWeight = Math.Max(1, baseWeight + recentPenalty - fatiguePenalty);

                for (int i = 0; i < finalWeight; i++)
                    pool.Add(candidate);
            }

            if (pool.Count == 0) return candidates.First();
            return pool[TaleWorlds.Core.MBRandom.RandomInt(pool.Count)];
        }
    }

    public static class VoiceResolver
    {
        public static string ResolveNarrative(string baseId, Warlord warlord, string extraParam1 = "", string extraParam2 = "")
        {
            // Find current warlord level (0 = Outlaw, 1 = Rebel, 2 = FamousBandit, 3 = Warlord, 4 = Recognized)
            int level = (int)WarlordLegitimacySystem.Instance.GetLevel(warlord.StringId);

            // 1. Try Kill-Count specific ID: e.g. {=BM_Nar_Agg_Kill10_1}
            if (baseId.EndsWith("_Kill") && int.TryParse(extraParam2, out int kills))
            {
                int[] thresholds = { 10, 3, 1 };
                foreach (int t in thresholds)
                {
                    if (kills >= t)
                    {
                        // Try variants 1, 2, 3
                        for (int v = 1; v <= 3; v++)
                        {
                            string killId = $"{baseId}{t}_{v}";
                            var killTextObj = new TextObject("{=" + killId + "}");
                            string resolvedKill = killTextObj.ToString();
                            if (!string.IsNullOrEmpty(resolvedKill) && resolvedKill != killId && !resolvedKill.Contains(baseId))
                            {
                                return $"[{warlord.Name}]: " + resolvedKill.Replace("{PLAYER}", extraParam1).Replace("{WARLORD}", warlord.Name);
                            }
                        }
                    }
                }
            }

            // 2. Try pulling from XML: e.g. {=BM_Nar_Agg_Kill_Lvl2}
            string idWithLevel = $"{baseId}_Lvl{level}";
            var textObjWithLevel = new TextObject("{=" + idWithLevel + "}");
            string resolvedWithLevel = textObjWithLevel.ToString();

            // If level-specific ID not found, try fallback (no level)
            if (string.IsNullOrEmpty(resolvedWithLevel) || resolvedWithLevel == idWithLevel || resolvedWithLevel.Contains(baseId))
            {
                var fallbackTextObj = new TextObject("{=" + baseId + "}FALLBACK");
                resolvedWithLevel = fallbackTextObj.ToString();

                // If XML fallback also missing, return C# hardcoded strings
                if (resolvedWithLevel == "FALLBACK")
                    return GetHardcodedFallback(baseId, warlord, level, extraParam1, extraParam2);
            }

            // Fill variables
            resolvedWithLevel = resolvedWithLevel
                .Replace("{WARLORD}", warlord.Name)
                .Replace("{PLAYER}", extraParam1)
                .Replace("{EXTRA}", extraParam2);

            return $"[{warlord.Name}]: {resolvedWithLevel}";
        }

        private static string GetHardcodedFallback(string baseId, Warlord warlord, int level, string param1, string param2)
        {
            // If no text found in XML either, return these backup texts. (COMBO MATRIX)
            if (baseId.Contains("Agg_Kill"))
            {
                if (level >= 3) return $"[{warlord.Name}]: My entire army hunts {param1}. Nowhere to hide!";
                return $"[{warlord.Name}]: You owe me a man, {param1}.";
            }
            if (baseId.Contains("Cau_Kill"))
            {
                if (level >= 3) return $"[{warlord.Name}]: The council is assembled. We will isolate and crush {param1}.";
                return $"[{warlord.Name}]: Fall back. {param1} has proven dangerous.";
            }
            if (baseId.Contains("Cun_Kill"))
            {
                if (level >= 3) return $"[{warlord.Name}]: Watch {param1} until we find a weakness.";
                return $"[{warlord.Name}]: Interesting tactic, {param1}. I've taken note.";
            }
            if (baseId.Contains("Ven_Kill"))
            {
                if (level >= 3) return $"[{warlord.Name}]: I've written your name in blood, {param1}! These lands will be your grave!";
                return $"[{warlord.Name}]: I will avenge my fallen brothers, {param1}.";
            }
            return $"[{warlord.Name}]: Target {param1} spotted.";
        }
    }

    public enum NpcVoiceType
    {
        Merchant,
        Soldier,
        Elder,
        Spy,
        Child,
        Bard
    }

    public static class WorldVoiceSynthesizer
    {
        // Dynamic lists (populated from XML at game load)
        private static readonly Dictionary<NpcVoiceType, List<string>> _prefixes = new();
        private static readonly List<string> _idioms = new();
        private static readonly List<string> _fallbackCors = new();
        private static bool _isInitialized = false;

        public static void InitializeFromXML()
        {
            if (_isInitialized) return;

            // Pull unlimited sentences from XML for 6 different NPC types
            var types = (NpcVoiceType[])Enum.GetValues(typeof(NpcVoiceType));
            foreach (var type in types)
            {
                _prefixes[type] = new List<string>();
                string typeKey = type.ToString().ToLower();
                int index = 1;

                while (true)
                {
                    // e.g.: {=bm_npc_merchant_1}
                    string id = $"{{=bm_npc_{typeKey}_{index}}}";
                    var textObj = new TaleWorlds.Localization.TextObject(id);
                    string localized = textObj.ToString();

                    // If no such ID exists in XML, Bannerlord returns the ID itself.
                    // Or returns empty. This tells us the array ended in XML!
                    if (string.IsNullOrEmpty(localized) || localized == id || localized.Contains($"bm_npc_{typeKey}"))
                        break; // Break loop, array ended!

                    _prefixes[type].Add(localized);
                    index++;
                }

                // If XML is missing (fallback) prevent crash
                if (_prefixes[type].Count == 0)
                {
                    _prefixes[type].Add(new TaleWorlds.Localization.TextObject($"{{=bm_npc_fallback_{typeKey}}}I'm just a villager, I don't know anything.").ToString());
                }
            }

            // Bard sayings / proverbs from XML
            int idiomIndex = 1;
            while (true)
            {
                string id = $"{{=bm_idiom_{idiomIndex}}}";
                var textObj = new TaleWorlds.Localization.TextObject(id);
                string localized = textObj.ToString();

                if (string.IsNullOrEmpty(localized) || localized == id || localized.Contains("bm_idiom_")) break;

                _idioms.Add(localized);
                idiomIndex++;
            }

            // Core events (will be shaped further with Combo_Matrix)
            int coreIndex = 1;
            while (true)
            {
                string id = $"{{=bm_core_event_{coreIndex}}}";
                var textObj = new TaleWorlds.Localization.TextObject(id);
                string localized = textObj.ToString();

                if (string.IsNullOrEmpty(localized) || localized == id || localized.Contains("bm_core_event_")) break;

                _fallbackCors.Add(localized);
                coreIndex++;
            }

            _isInitialized = true;
        }

        public static string GenerateAmbientLine(Warlord warlord, NarrativeMemory memory)
        {
            if (!_isInitialized) InitializeFromXML();

            var types = (NpcVoiceType[])Enum.GetValues(typeof(NpcVoiceType));
            var type = types[MBRandom.RandomInt(types.Length)];

            string prefix = GetPrefix(type, memory);
            string core = GetCoreEvent(warlord, memory);
            string idiom = GetIdiom(memory);

            // Random idiom addition (30%)
            bool addIdiom = MBRandom.RandomFloat < 0.3f && !memory.IsRecentlyUsed(idiom);

            string finalMessage = $"{GetTypeLabel(type)}: \"{prefix} {core}";
            if (addIdiom)
            {
                finalMessage += $" {idiom}\"";
                memory.RecordEvent(idiom, memory.DaysSinceEscalation);
            }
            else
            {
                finalMessage += "\"";
            }

            memory.RecordEvent(prefix, memory.DaysSinceEscalation);
            memory.RecordEvent(core, memory.DaysSinceEscalation);

            return finalMessage;
        }

        private static string GetPrefix(NpcVoiceType type, NarrativeMemory memory)
        {
            if (!_prefixes.TryGetValue(type, out var list)) return "";
            return memory.WeightedPick(list, x => x);
        }

        private static string GetCoreEvent(Warlord warlord, NarrativeMemory memory)
        {
            string raw = memory.WeightedPick(_fallbackCors, x => x);
            return raw.Replace("{WARLORD}", warlord.Name);
        }

        private static string GetIdiom(NarrativeMemory memory)
        {
            return memory.WeightedPick(_idioms, x => x);
        }

        private static string GetTypeLabel(NpcVoiceType type)
        {
            return type switch
            {
                NpcVoiceType.Merchant => new TextObject("{=BM_NPC_Merchant}Merchant").ToString(),
                NpcVoiceType.Soldier => new TextObject("{=BM_NPC_Soldier}Soldier").ToString(),
                NpcVoiceType.Elder => new TextObject("{=BM_NPC_Elder}Elder").ToString(),
                NpcVoiceType.Spy => new TextObject("{=BM_NPC_Spy}Spy").ToString(),
                NpcVoiceType.Child => new TextObject("{=BM_NPC_Child}Child").ToString(),
                NpcVoiceType.Bard => new TextObject("{=BM_NPC_Bard}Bard").ToString(),
                _ => new TextObject("{=BM_NPC_Villager}Villager").ToString()
            };
        }
    }
